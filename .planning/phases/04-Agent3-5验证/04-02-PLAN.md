@pytest.mark.parametrize("extracted_text,min_expected_level", [
    ("立即转账 今天必须 不转就走人", "高"),           # R001 紧急转账
    ("公安局 检察院 安全账户 配合调查", "高"),         # R002 冒充公检法
    ("刷单 刷好评 兼职接单 日赚佣金", "高"),           # R003 刷单
    ("恭喜中奖 领取奖品 交保证金 手续费", "高"),       # R006 中奖
])
def test_high_risk_keywords_trigger_high_level(agent, step_dir, extracted_text, min_expected_level)：
   - step1 = {task_id:f"kw-test-{i}", intent_label:"fraud", extracted_text:extracted_text}
   - step2 = {avg_similarity:0.9, low_similarity_warning:False}
   - step3 = {fraud_probability:0.80, verdict:"诈骗", fraud_type:"测试"}
   - result = agent.run(step1, step2, step3, step_dir)
   - LEVEL_ORDER = ["安全", "低", "中", "高", "极高"]
   - actual_idx = LEVEL_ORDER.index(result["risk_level"])
   - expected_idx = LEVEL_ORDER.index(min_expected_level)
   - assert actual_idx >= expected_idx, f"期望 >= {min_expected_level}, 实际 {result['risk_level']} (score={result['final_risk_score']:.2f})"

test_extreme_high_requires_guardian_alert(agent, step_dir)：
   - step3 = {fraud_probability:0.95, verdict:"诈骗", fraud_type:"极高风险测试"}（base_score=0.95 >= 0.90 阈值）
   - step1 = {task_id:"guardian-test", intent_label:"fraud", extracted_text:"立即转账 公安局 安全账户"}
   - step2 = {avg_similarity:0.9}
   - result = agent.run(step1, step2, step3, step_dir)
   - 若 result["risk_level"] == "极高"：assert result["requires_guardian_alert"] is True
   - 否则：pytest.skip(f"分数不足以触发极高: {result['final_risk_score']:.3f}")

test_whitelist_rule_reduces_score(agent, step_dir)：
   - step3_high = {fraud_probability:0.70}（中高区间）
   - step1_with_whitelist = {extracted_text:"保险产品 理财规划 定期存款", intent_label:"normal"}

   - result_no = agent.run(step1_no_whitelist, step2, step3_high, step_dir)
   - assert result_white["final_risk_score"] <= result_no["final_risk_score"]，"白名单词应降低分数"
